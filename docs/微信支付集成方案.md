# 微信支付集成方案

## 一、添加依赖

在 `pom.xml` 中添加微信支付SDK依赖：

```xml
<!-- 微信支付SDK -->
<dependency>
    <groupId>com.github.wechatpay-apiv3</groupId>
    <artifactId>wechatpay-java</artifactId>
    <version>0.2.12</version>
</dependency>
```

## 二、配置微信支付参数

在 `application.yml` 中添加配置：

```yaml
# 微信支付配置
wechat:
  pay:
    app-id: your-app-id          # 微信AppID
    mch-id: your-mch-id          # 商户号
    api-key: your-api-key        # API密钥
    cert-path: classpath:cert/apiclient_cert.p12  # 证书路径
    notify-url: http://your-domain.com/api/order/wechat-notify  # 支付回调地址
```

## 三、创建微信支付服务类

创建 `WechatPayService.java`：

```java
package com.overseas.purchase.service;

import com.wechat.pay.java.core.Config;
import com.wechat.pay.java.core.RSAAutoCertificateConfig;
import com.wechat.pay.java.service.payments.jsapi.JsapiService;
import com.wechat.pay.java.service.payments.jsapi.model.*;
import lombok.RequiredArgsConstructor;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Service;

import java.math.BigDecimal;

@Service
@RequiredArgsConstructor
public class WechatPayService {
    
    @Value("${wechat.pay.app-id}")
    private String appId;
    
    @Value("${wechat.pay.mch-id}")
    private String mchId;
    
    @Value("${wechat.pay.api-key}")
    private String apiKey;
    
    @Value("${wechat.pay.cert-path}")
    private String certPath;
    
    @Value("${wechat.pay.notify-url}")
    private String notifyUrl;
    
    private JsapiService jsapiService;
    
    /**
     * 初始化微信支付服务
     */
    private void initService() {
        if (jsapiService == null) {
            // 使用自动更新证书的配置
            Config config = new RSAAutoCertificateConfig.Builder()
                    .merchantId(mchId)
                    .privateKeyFromPath(certPath)
                    .merchantSerialNumber("your-serial-number")
                    .apiV3Key(apiKey)
                    .build();
            
            jsapiService = new JsapiService.Builder().config(config).build();
        }
    }
    
    /**
     * 创建JSAPI支付订单
     */
    public PrepayResponse createJsapiOrder(String orderNo, BigDecimal amount, String description, String openId) {
        initService();
        
        PrepayRequest request = new PrepayRequest();
        request.setAppid(appId);
        request.setMchid(mchId);
        request.setDescription(description);
        request.setOutTradeNo(orderNo);
        request.setNotifyUrl(notifyUrl);
        
        Amount amountObj = new Amount();
        amountObj.setTotal(amount.multiply(new BigDecimal(100)).intValue()); // 转换为分
        amountObj.setCurrency("CNY");
        request.setAmount(amountObj);
        
        Payer payer = new Payer();
        payer.setOpenid(openId);
        request.setPayer(payer);
        
        return jsapiService.prepay(request);
    }
    
    /**
     * 查询订单支付状态
     */
    public Transaction queryOrder(String orderNo) {
        initService();
        return jsapiService.queryOrderByOutTradeNo(new QueryOrderByOutTradeNoRequest.Builder()
                .outTradeNo(orderNo)
                .mchid(mchId)
                .build());
    }
}
```

## 四、修改订单服务

在 `OrderService.java` 中添加微信支付相关方法：

```java
// 注入微信支付服务
private final WechatPayService wechatPayService;

/**
 * 创建微信支付订单
 */
public Map<String, Object> createWechatPayOrder(Long orderId, String openId) {
    Order order = orderMapper.selectById(orderId);
    if (order == null || order.getDeleted() == 1) {
        throw new RuntimeException("订单不存在");
    }
    
    if (!"PENDING_PAYMENT".equals(order.getStatus())) {
        throw new RuntimeException("订单状态不正确");
    }
    
    // 调用微信支付创建订单
    PrepayResponse prepayResponse = wechatPayService.createJsapiOrder(
        order.getOrderNo(),
        order.getTotalPrice(),
        "海外代购订单-" + order.getOrderNo(),
        openId
    );
    
    // 返回前端需要的支付参数
    Map<String, Object> result = new HashMap<>();
    result.put("prepayId", prepayResponse.getPrepayId());
    result.put("appId", wechatPayService.getAppId());
    result.put("timeStamp", String.valueOf(System.currentTimeMillis() / 1000));
    result.put("nonceStr", prepayResponse.getNonceStr());
    result.put("package", "prepay_id=" + prepayResponse.getPrepayId());
    result.put("signType", "RSA");
    
    // 生成签名（需要实现签名算法）
    String paySign = generatePaySign(result);
    result.put("paySign", paySign);
    
    return result;
}

/**
 * 处理微信支付回调
 */
@Transactional
public void handleWechatPayNotify(Map<String, Object> notifyData) {
    // 验证签名
    // 处理支付结果
    String orderNo = (String) notifyData.get("out_trade_no");
    String tradeState = (String) notifyData.get("trade_state");
    
    if ("SUCCESS".equals(tradeState)) {
        Order order = orderMapper.selectOne(
            new LambdaQueryWrapper<Order>()
                .eq(Order::getOrderNo, orderNo)
        );
        
        if (order != null && "PENDING_PAYMENT".equals(order.getStatus())) {
            order.setStatus("PENDING_SHIPMENT");
            orderMapper.updateById(order);
        }
    }
}
```

@PostMapping("/wechat-pay/{id}")
public Result<Map<String, Object>> createWechatPay(@PathVariable Long id, 
                                                     @RequestBody Map<String, Object> params,

## 六、修改前端支付逻辑

在 `Orders.vue` 中修改支付方法：

```javascript
async payOrder(id) {
  try {
    // 获取用户openId（需要先调用微信授权）
    const openId = await this.getWechatOpenId()
    
    // 创建微信支付订单
    const payRes = await orderApi.createWechatPay(id, { openId })
    const payParams = payRes.data
    
    // 调用微信支付
    if (window.WeixinJSBridge) {
      window.WeixinJSBridge.invoke(
        'getBrandWCPayRequest',
        {
          appId: payParams.appId,
          timeStamp: payParams.timeStamp,
          nonceStr: payParams.nonceStr,
          package: payParams.package,
          signType: payParams.signType,
          paySign: payParams.paySign
        },
        (res) => {
          if (res.err_msg === 'get_brand_wcpay_request:ok') {
            this.$message.success('支付成功')
            this.loadOrders()
          } else {
            this.$message.error('支付失败')
          }
        }
      )
    } else {
      this.$message.error('请在微信中打开')
    }
  } catch (error) {
    this.$message.error(error.message || '支付失败')
  }
}
```

## 七、注意事项

1. **证书配置**：需要将微信支付证书文件放在 `src/main/resources/cert/` 目录下
2. **回调地址**：需要在微信商户平台配置支付回调地址
3. **域名配置**：需要在微信公众平台配置JS接口安全域名
4. **测试环境**：可以使用微信支付沙箱环境进行测试
5. **签名算法**：需要实现微信支付的签名算法（RSA）

## 八、简化方案（推荐用于毕业设计）

如果只是毕业设计演示，可以：
1. 保留当前的模拟支付功能
2. 添加一个"微信支付"按钮，点击后显示支付二维码（使用微信支付统一下单API生成）
3. 或者使用微信支付的H5支付，跳转到微信支付页面

这样可以展示支付流程，但不需要完整的微信支付集成。
